from flask import Flask
import threading
import os

app = Flask(__name__)

@app.route('/')
def home():
    return "Bot is alive!"

# サーバー停止用のエンドポイント
@app.route('/shutdown', methods=['POST'])
def shutdown():
    os._exit(0)  # プロセスを完全終了
    return "Shutting down...", 200

def run():
    app.run(host="0.0.0.0", port=8080)

def keep_alive():
    server = threading.Thread(target=run)
    server.start()
